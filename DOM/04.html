public class UniqueNumber { static int findUnique(int[] arr) { int n = arr.length; // Mark duplicates as -1 for (int i = 0; i < n; i++) { for (int j = i + 1; j < n; j++) { if (arr[i] == arr[j] && arr[i] != -1) { arr[i] = -1; arr[j] = -1; } } } // The number not marked as -1 is unique int ans = -1; for (int i = 0; i < n; i++) { if (arr[i] != -1) { ans = arr[i]; break; } } return ans; } public static void main(String[] args) { Scanner sc = new Scanner(System.in); int n = sc.nextInt(); int[] arr = new int[n]; System.out.println("Enter " + n + " elements"); for (int i = 0; i < n; i++) { arr[i] = sc.nextInt(); } System.out.println("unique element: " + findUnique(arr)); } }
